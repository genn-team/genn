254  template<typename T, typename S, typename F>
255  void genParallelGroup(CodeStream &os, const Substitutions &kernelSubs, const std::vector<T> &groups, size_t &idStart,
256  S getPaddedSizeFunc, F filter, GroupHandler<T> handler) const
257  {
258  // Loop through groups
259  for(const auto &gMerge : groups) {
260  if(filter(gMerge)) {
261  // Sum padded sizes of each group within merged group
262  const size_t paddedSize = std::accumulate(
263  gMerge.getGroups().cbegin(), gMerge.getGroups().cend(), size_t{0},
264  [gMerge, getPaddedSizeFunc](size_t acc, std::reference_wrapper<const typename T::GroupInternal> g)
265  {
266  return (acc + getPaddedSizeFunc(g.get()));
267  });
268 
269  os << "// merged" << gMerge.getIndex() << std::endl;
270 
271  // If this is the first group
272  if(idStart == 0) {
273  os << "if(id < " << paddedSize << ")";
274  }
275  else {
276  os << "if(id >= " << idStart << " && id < " << idStart + paddedSize << ")";
277  }
278  {
279  CodeStream::Scope b(os);
280  Substitutions popSubs(&kernelSubs);
281 
282  if(gMerge.getGroups().size() == 1) {
283  os << getPointerPrefix() << "struct Merged" << T::name << "Group" << gMerge.getIndex() << " *group";
284  os << " = &d_merged" << T::name << "Group" << gMerge.getIndex() << "[0]; " << std::endl;
285  os << "const unsigned int lid = id - " << idStart << ";" << std::endl;
286 
287  // Use the starting thread ID of the whole merged group as group_start_id
288  popSubs.addVarSubstitution("group_start_id", std::to_string(idStart));
289  }
290  else {
291  // Perform bisect operation to get index of merged struct
292  os << "unsigned int lo = 0;" << std::endl;
293  os << "unsigned int hi = " << gMerge.getGroups().size() << ";" << std::endl;
294  os << "while(lo < hi)" << std::endl;
295  {
296  CodeStream::Scope b(os);
297  os << "const unsigned int mid = (lo + hi) / 2;" << std::endl;
298 
299  os << "if(id < d_merged" << T::name << "GroupStartID" << gMerge.getIndex() << "[mid])";
300  {
301  CodeStream::Scope b(os);
302  os << "hi = mid;" << std::endl;
303  }
304  os << "else";
305  {
306  CodeStream::Scope b(os);
307  os << "lo = mid + 1;" << std::endl;
308  }
309  }
310 
311  // Use this to get reference to merged group structure
312  os << getPointerPrefix() << "struct Merged" << T::name << "Group" << gMerge.getIndex() << " *group";
313  os << " = &d_merged" << T::name << "Group" << gMerge.getIndex() << "[lo - 1]; " << std::endl;
314 
315  // Get group start thread ID and use as group_start_id
316  os << "const unsigned int groupStartID = d_merged" << T::name << "GroupStartID" << gMerge.getIndex() << "[lo - 1];" << std::endl;
317  popSubs.addVarSubstitution("group_start_id", "groupStartID");
318 
319  // Use this to calculate local id within group
320  os << "const unsigned int lid = id - groupStartID;" << std::endl;
321  }
322  popSubs.addVarSubstitution("id", "lid");
323 
324  handler(os, gMerge, popSubs);
325 
326  idStart += paddedSize;
327  }
328  }
329  }
