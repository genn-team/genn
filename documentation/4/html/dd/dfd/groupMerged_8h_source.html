181  template<typename T, typename G, typename C>
182  void orderNeuronGroupChildren(const std::vector<T> &archetypeChildren,
183  std::vector<std::vector<T>> &sortedGroupChildren,
184  G getVectorFunc, C isCompatibleFunc) const
185  {
186  // Reserve vector of vectors to hold children for all neuron groups, in archetype order
187  sortedGroupChildren.reserve(archetypeChildren.size());
188 
189  // Loop through groups
190  for(const auto &g : getGroups()) {
191  // Make temporary copy of this group's children
192  std::vector<T> tempChildren((g.get().*getVectorFunc)());
193 
194  assert(tempChildren.size() == archetypeChildren.size());
195 
196  // Reserve vector for this group's children
197  sortedGroupChildren.emplace_back();
198  sortedGroupChildren.back().reserve(tempChildren.size());
199 
200  // Loop through archetype group's children
201  for(const auto &archetypeG : archetypeChildren) {
202  // Find compatible child in temporary list
203  const auto otherChild = std::find_if(tempChildren.cbegin(), tempChildren.cend(),
204  [archetypeG, isCompatibleFunc](const T &g)
205  {
206  return isCompatibleFunc(archetypeG, g);
207  });
208  assert(otherChild != tempChildren.cend());
209 
210  // Add pointer to vector of compatible merged in syns
211  sortedGroupChildren.back().push_back(*otherChild);
212 
213  // Remove from original vector
214  tempChildren.erase(otherChild);
215  }
216  }
217  }
