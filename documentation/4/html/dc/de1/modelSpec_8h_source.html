529  template<typename PostsynapticModel>
530  SynapseGroup *addSlaveSynapsePopulation(const std::string &name, const std::string &weightSharingMasterName, unsigned int delaySteps, const std::string &src, const std::string &trg,
531  const PostsynapticModel *psm, const typename PostsynapticModel::ParamValues &postsynapticParamValues, const typename PostsynapticModel::VarValues &postsynapticVarInitialisers)
532  {
533  // Get source and target neuron groups
534  auto srcNeuronGrp = findNeuronGroupInternal(src);
535  auto trgNeuronGrp = findNeuronGroupInternal(trg);
536 
537  // Find weight sharing master group
538  auto masterGrp = findSynapseGroupInternal(weightSharingMasterName);
539  const auto *wum = masterGrp->getWUModel();
540 
541  // If the weight sharing master has individuak weights and any are read-write, give error
542  const auto wumVars = wum->getVars();
543  if((masterGrp->getMatrixType() & SynapseMatrixWeight::INDIVIDUAL) &&
544  std::any_of(wumVars.cbegin(), wumVars.cend(),
545  [](const Models::Base::Var &v)
546  {
547  return (v.access == VarAccess::READ_WRITE);
548  }))
549  {
550  throw std::runtime_error("Individual synapse variables can only be shared if they are read-only");
551  }
552 
553  // Check that population sizes match
554  if ((srcNeuronGrp->getNumNeurons() != masterGrp->getSrcNeuronGroup()->getNumNeurons())
555  || (trgNeuronGrp->getNumNeurons() != masterGrp->getTrgNeuronGroup()->getNumNeurons()))
556  {
557  throw std::runtime_error("Size of populations connected by shared weights must match");
558  }
559 
560  // If weight update model has any pre or postsynaptic variables, give error
561  // **THINK** this could be supported but quite what the semantics are is ambiguous
562  if(!wum->getPreVars().empty() || !wum->getPostVars().empty()) {
563  throw std::runtime_error("Synapse groups with pre and postsynpatic variables cannot be shared");
564  }
565 
566  // Add synapse group to map
567  auto uninitialisedToeplitz = InitToeplitzConnectivitySnippet::Init(InitToeplitzConnectivitySnippet::Uninitialised::getInstance(), {});
568  auto result = m_LocalSynapseGroups.emplace(
569  std::piecewise_construct,
570  std::forward_as_tuple(name),
571  std::forward_as_tuple(name, masterGrp, masterGrp->getMatrixType(), delaySteps,
572  wum, masterGrp->getWUParams(), masterGrp->getWUVarInitialisers(), masterGrp->getWUPreVarInitialisers(), masterGrp->getWUPostVarInitialisers(),
573  psm, postsynapticParamValues.getInitialisers(), postsynapticVarInitialisers.getInitialisers(),
574  srcNeuronGrp, trgNeuronGrp, masterGrp->getConnectivityInitialiser(), uninitialisedToeplitz,
575  m_DefaultVarLocation, m_DefaultExtraGlobalParamLocation, m_DefaultSparseConnectivityLocation, m_DefaultNarrowSparseIndEnabled));
576 
577  if(!result.second) {
578  throw std::runtime_error("Cannot add a synapse population with duplicate name:" + name);
579  }
580  else {
581  return &result.first->second;
582  }
583  }
