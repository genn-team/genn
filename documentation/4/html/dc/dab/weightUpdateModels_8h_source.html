308  SET_PARAM_NAMES({"tLrn", "tChng", "tDecay", "tPunish10", "tPunish01",
309  "gMax", "gMid", "gSlope", "tauShift", "gSyn0"});
310  SET_VARS({{"g", "scalar"}, {"gRaw", "scalar"}});
311 
312  SET_SIM_CODE(
313  "$(addToInSyn, $(g));\n"
314  "scalar dt = $(sT_post) - $(t) - ($(tauShift)); \n"

322  "else dg = - ($(off2)) ; \n"

327  "scalar dg =0; \n"
328  "if (dt > $(lim0)) \n"
329  " dg = -($(off0)) ; \n"
330  "else if (dt > 0) \n"

332  "else if (dt > $(lim1)) \n"
