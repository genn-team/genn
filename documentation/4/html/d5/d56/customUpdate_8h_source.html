99  void checkVarReferences(const std::vector<V> &varRefs)
100  {
101  const auto modelVarRefs = getCustomUpdateModel()->getVarRefs();
102 
103  // If target of any variable references is duplicated, custom update should be batched
104  m_Batched = std::any_of(varRefs.cbegin(), varRefs.cend(),
105  [](const V &v) { return (v.getVar().access & VarAccessDuplication::DUPLICATE); });
106 
107  // Loop through all variable references
108  for(size_t i = 0; i < varRefs.size(); i++) {
109  const auto varRef = varRefs.at(i);
110  const auto modelVarRef = modelVarRefs.at(i);
111 
112  // Check types of variable references against those specified in model
113  // **THINK** due to GeNN's current string-based type system this is rather conservative
114  if(varRef.getVar().type != modelVarRef.type) {
115  throw std::runtime_error("Incompatible type for variable reference '" + modelVarRef.name + "'");
116  }
117 
118  // Check that no reduction targets reference duplicated variables
119  if((varRef.getVar().access & VarAccessDuplication::DUPLICATE)
120  && (modelVarRef.access & VarAccessModeAttribute::REDUCE))
121  {
122  throw std::runtime_error("Reduction target variable reference must be to SHARED variables.");
123  }
124 
125  // If custom update is batched, check that any variable references to shared variables are read-only
126  // **NOTE** if custom update isn't batched, it's totally fine to write to shared variables
127  if(m_Batched && (varRef.getVar().access & VarAccessDuplication::SHARED)
128  && (modelVarRef.access != VarAccessMode::READ_ONLY))
129  {
130  throw std::runtime_error("Variable references to SHARED variables in batched custom updates must be read-only.");
131  }
132  }
133  }
