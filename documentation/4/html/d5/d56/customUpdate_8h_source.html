95  bool isReduction(const std::vector<V> &varRefs, VarAccessDuplication duplication) const
96  {
97  // Return true if any variables have REDUCE flag in their access mode and have correct duplication flag
98  const auto vars = getCustomUpdateModel()->getVars();
99  if(std::any_of(vars.cbegin(), vars.cend(),
100  [duplication](const Models::Base::Var &v)
101  {
102  return (v.access & VarAccessModeAttribute::REDUCE) && (v.access & duplication);
103  }))
104  {
105  return true;
106  }
107 
108  // Loop through all variable references
109  const auto modelVarRefs = getCustomUpdateModel()->getVarRefs();
110  for (size_t i = 0; i < varRefs.size(); i++) {
111  const auto varRef = varRefs.at(i);
112  const auto modelVarRef = modelVarRefs.at(i);
113 
114  // If custom update model reduces into this variable reference and the variable it targets has correct duplication flag
115  if ((modelVarRef.access & VarAccessModeAttribute::REDUCE) & (varRef.getVar().access & duplication)) {
116  return true;
117  }
118  }
119 
120  return false;
121  }

128  const auto modelVarRefs = getCustomUpdateModel()->getVarRefs();
129  for(size_t i = 0; i < varRefs.size(); i++) {

149  template<typename V>
150  void checkVarReferenceBatching(const std::vector<V>& varRefs, unsigned int batchSize)
151  {
152  // If target of any variable references is not shared across batches, custom update should be batched
153  if(batchSize > 1) {
154  m_Batched = std::any_of(varRefs.cbegin(), varRefs.cend(),
155  [](const V& v)
156  {
157  return (v.isBatched() && !(v.getVar().access & VarAccessDuplication::SHARED));
158  });
159  }
160  else {
161  m_Batched = false;
162  }
163 
164  // Loop through all variable references
165  const auto modelVarRefs = getCustomUpdateModel()->getVarRefs();
166  for (size_t i = 0; i < varRefs.size(); i++) {
167  const auto varRef = varRefs.at(i);
168  const auto modelVarRef = modelVarRefs.at(i);
169 
170  // If custom update is batched, check that any variable references to shared variables are read-only
171  // **NOTE** if custom update isn't batched, it's totally fine to write to shared variables
172  if(m_Batched && (varRef.getVar().access & VarAccessDuplication::SHARED)
173  && (modelVarRef.access == VarAccessMode::READ_WRITE))
174  {
175  throw std::runtime_error("Variable references to SHARED variables in batched custom updates cannot be read-write.");
176  }
177  }
