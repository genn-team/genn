250  }

254  {

256  this->isParamValueHeterogeneous(paramIndex, [varIndex](const G &cg) { return cg.getVarInitialisers().at(varIndex).getDerivedParams(); }));

259 protected:
260  CustomUpdateInitGroupMergedBase(size_t index, const std::string &precision, const BackendBase &backend,
261  const std::vector<std::reference_wrapper<const G>> &groups)
262  : GroupMerged<G>(index, precision, groups)
263  {
264  // Loop through variables
265  const CustomUpdateModels::Base *cm = this->getArchetype().getCustomUpdateModel();
266  const auto vars = cm->getVars();
267  const auto &varInit = this->getArchetype().getVarInitialisers();
268  assert(vars.size() == varInit.size());
269  for(size_t v = 0; v < vars.size(); v++) {
270  // If we're not initialising or if there is initialization code for this variable
271  const auto var = vars[v];
272  if(!varInit[v].getSnippet()->getCode().empty()) {
273  this->addPointerField(var.type, var.name, backend.getDeviceVarPrefix() + var.name);
274  }
275 
276  // Add any var init EGPs to structure
277  this->addEGPs(varInit[v].getSnippet()->getExtraGlobalParams(), backend.getDeviceVarPrefix(), var.name);
278  }
279 
280  this->template addHeterogeneousVarInitParams<CustomUpdateInitGroupMergedBase<G>>(
281  vars, &G::getVarInitialisers,
282  &CustomUpdateInitGroupMergedBase<G>::isVarInitParamHeterogeneous);
283 
284  this->template addHeterogeneousVarInitDerivedParams<CustomUpdateInitGroupMergedBase<G>>(
285  vars, &G::getVarInitialisers,
286  &CustomUpdateInitGroupMergedBase<G>::isVarInitDerivedParamHeterogeneous);
287  }

293  {
294  // Update hash with archetype's hash digest

311  {
312  // If parameter isn't referenced in code, there's no point implementing it hetereogeneously!
313  const auto *varInitSnippet = this->getArchetype().getVarInitialisers().at(varIndex).getSnippet();
314  const std::string paramName = varInitSnippet->getParamNames().at(paramIndex);
