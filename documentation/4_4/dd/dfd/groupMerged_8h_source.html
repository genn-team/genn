74  {

135  size_t getStructArraySize(const BackendBase &backend) const
136  {
137  // Loop through fields again to generate any EGP pushing functions that are required and to calculate struct size
138  size_t structSize = 0;
139  size_t largestFieldSize = 0;
140  const auto sortedFields = getSortedFields(backend);
141  for(const auto &f : sortedFields) {
142  // Add size of field to total
143  const size_t fieldSize = backend.getSize(std::get<0>(f));
144  structSize += fieldSize;
145 
146  // Update largest field size

320  {
321  // Make a copy of fields and sort so largest come first. This should mean that due
322  // to structure packing rules, significant memory is saved and estimate is more precise
323  auto sortedFields = getSortedFields(backend);
324 
325  // If this isn't a host merged structure, generate definition for function to push group
326  if(!host) {
327  definitionsInternalFunc << "EXPORT_FUNC void pushMerged" << name << "Group" << getIndex() << "ToDevice(unsigned int idx, ";
328  generateStructFieldArgumentDefinitions(definitionsInternalFunc, backend);

341  assert(std::get<3>(f) != FieldType::Host || host);

345  if(host) {

348  generateStruct(definitionsInternal, backend, name, true);

355  }

490  G getVectorFunc, C isCompatibleFunc) const
491  {
492  // Reserve vector of vectors to hold children for all neuron groups, in archetype order
493  sortedGroupChildren.reserve(archetypeChildren.size());
494

513  });
514  assert(otherChild != tempChildren.cend());

517  sortedGroupChildren.back().push_back(*otherChild);
518 
519  // Remove from original vector
